# 常用软件安装

# mysql

  1. 下载mysql

    docker pull mysql

  2. 启动容器

    docker run -d -p 3307:3306 --privileged=true \
-e MYSQL_ROOT_PASSWORD=123456 \
-v /docker/mysql7/log:/var/log/mysql \
-v /docker/mysql7/data:/var/lib/mysql \
-v /docker/mysql7/conf:/etc/mysql/conf.d \
-v /etc/localtime:/etc/localtime \
--name mysql7 \
--restart=always \
mysql

# mysql主从复制

  1. 创建mysql-master容器

    • 配置文件 my.cnf

      [mysqld]
pid-file        = /var/run/mysqld/mysqld.pid
socket          = /var/run/mysqld/mysqld.sock
datadir         = /var/lib/mysql
secure-file-priv= NULL

server_id=101
binlog-ignore-db=mysql
log-bin=mall-mysql-bin
binlog_cache_size=1M
binlog_format=mixed
expire_logs_days=7
slave_skip_errors=1062

# Custom config should go here
!includedir /etc/mysql/conf.d/

    • 在执行启动脚本之前，将my.cnf文件放置到/docker/mysql-master/conf文件夹下，并在/docker/mysql-master/conf下创建conf.d文件夹

    • 启动脚本

        docker run -d -p 3307:3306 --privileged=true \
  -e MYSQL_ROOT_PASSWORD=123456 \
  -v /docker/mysql-master/log:/var/log/mysql \
  -v /docker/mysql-master/data:/var/lib/mysql \
  -v /docker/mysql-master/conf:/etc/mysql/ \
  -v /etc/localtime:/etc/localtime \
  --name mysql-master \
  --restart=always \
  mysql

    • 进入mysql-master

    • 创建同步用户

      create user 'slave'@'%'identified by '123456';
grant replication slave,replication client on *.* to 'slave'@'%';
flush privileges;

    • 查看同步状态

      show master status;

      记录下file和position

  2. 创建mysql-slave容器

    • 配置文件

      [mysqld]
pid-file        = /var/run/mysqld/mysqld.pid
socket          = /var/run/mysqld/mysqld.sock
datadir         = /var/lib/mysql
secure-file-priv= NULL

server_id=102
binlog-ignore-db=mysql
log-bin=mall-mysql-slave1-bin
binlog_cache_size=1M
binlog_format=mixed
expire_logs_days=7
slave_skip_errors=1062
relay_log=mall-mysql-relay-bin
log_slave_updates=1
#只对没有super权限存在限制，使用时需要从库创建并使用普通用户。GRANT USAGE ON *.* TO 'user01'@'localhost' IDENTIFIED BY'123456'WITH GRANT OPTION;
read_only=1 

# Custom config should go here
!includedir /etc/mysql/conf.d/

    • 将my.cnf文件放置在/docker/mysql-slave/conf路径下，并创建conf.d文件夹

    • 启动脚本

      docker run -d -p 3308:3306 --privileged=true \
-e MYSQL_ROOT_PASSWORD=123456 \
-v /docker/mysql-slave/log:/var/log/mysql \
-v /docker/mysql-slave/data:/var/lib/mysql \
-v /docker/mysql-slave/conf:/etc/mysql \
-v /etc/localtime:/etc/localtime \
--name mysql-slave \
--restart=always \
mysql

    • 从服务器配置主从复制

      1. 这个命令在mysql8上会导致从服务器无法连接到主服务器，因为8.0更新了账户登录验证插件
      change master to master_host='192.168.142.132',
master_user='slave',
master_password='123456',
master_port=3307,
master_log_file='mall-mysql-bin.000010',
master_log_pos=711,
master_connect_retry=30;

      1. 8.0可使用的命令

        CHANGE REPLICATION SOURCE TO 
GET_SOURCE_PUBLIC_KEY=1, 
source_host='192.168.142.132',
source_user='slave',
source_password='123456',
source_port=3307,
source_log_file='mall-mysql-bin.000010',
source_log_pos=711,
source_connect_retry=30;

    • 启动主从复制

      start slave;

    • 查看从机主从复制状态

      show slave status \G;

      # 从库删除数据重新同步

      1. 停止slave,清除主从信息

        stop slave;
reset slave all;

      2. 删除从库同步的主数据库数据

      3. 备份主数据库数据

        reset master;
mysqldump -u root -p test --set-gtid-purged=OFF --single-transaction > dbname.sql

      4. 从库导入数据

      5. 从库配置主从复制

        CHANGE REPLICATION SOURCE TO 
GET_SOURCE_PUBLIC_KEY=1, 
source_host='192.168.142.132',
source_user='slave',
source_password='123456',
source_port=3307,
source_log_file='mall-mysql-bin.000001',
source_log_pos=156,
source_connect_retry=30;

      6. 启动slave

      7. 查看slave状态

# 扩容

添加master节点

docker run -d --name redis-node-7 --net host -v /docker/redis/redis-node-7:/data redis:6.2.6 --cluster-enabled yes --appendonly yes --port 6387

#redis-cli --cluster add-node 新节点ip:端口 集群ip:端口
redis-cli --cluster add-node 192.168.142.132:6387 192.168.142.132:6381

redis-cli --cluster reshard 192.168.142.132:6381

添加slave节点

docker run -d --name redis-node-8 --net host -v /docker/redis/redis-node-8:/data redis:6.2.6 --cluster-enabled yes --appendonly yes --port 6388

# redis-cli --cluster add-node 新slave的ip:prot 新master的ip:port --cluster-slave --cluster-master-id 新主机节点id
redis-cli --cluster add-node 192.168.142.132:6388 192.168.142.132:6387 --cluster-slave --cluster-master-id fb51d858020dd29dd39aecac3593b9317d7cd091

# 缩容

  1. 删除slave节点

    #redis-cli --cluster del-node ip:port slave的id
redis-cli --cluster del-node 192.168.142.132:6388 f48983ca68b878f9282b16270a390f0f11e05c37

  2. 重新分配槽

    redis-cli --cluster reshard 192.168.142.132:6381

  3. 删除master节点

    redis-cli --cluster del-node 192.168.142.132:6387 fb51d858020dd29dd39aecac3593b9317d7cd091
