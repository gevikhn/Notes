# 常用命令

# 镜像仓库命令

  1. login

  2. logout

  3. pull：下载镜像

  4. push

  5. search :查找镜像

    docker search nginx
# STARS 关注度
docker search --filter=stars=300 nginx #查找关注度大于300的镜像

# 镜像命令

  1. build

  2. images：查看本地镜像

    docker images -a            #列出本地所有的镜像
docker images -q            #只显示镜像ID
docker images --digests     #显示镜像的摘要信息
docker images --no-trunc    #显示完整的镜像信息

  3. import

  4. load

  5. rmi：删除镜像

  6. save

  7. tag

  8. commit

# 容器生命周期管理

  1. create

  2. exec：进入容器，退出时不会停止容器

    docker exec -it MyUbuntu /bin/bash

  3. kill：停止容器

  4. pause

  5. restart：重启容器

  6. rm：删除容器

  7. run：启动一个新容器，-i 以交互模式运行容器，-t 分配一个伪终端，-d 后台运行，-p 分配端口，-P随机分配端口，-v 挂载数据卷，-c 指定容器分配cpu的shares值，-m 限制容器的内存信息，以B、K、M、G为单位，--name 指定名称，--net 容器使用的网络

    docker run -itd --name MyUbuntu ubuntu:20.10 /bin/bash
docker run -itd -p 8080:80 nginx

  8. start：启动容器

  9. stop：停止容器

  10. unpause

# 容器运维操作

  1. attach：进入容器，退出时会停止容器

  2. export

  3. inspect

  4. port

  5. ps：查看容器

    # 查看正在运行的容器
docker ps
# 查看所有容器
docker ps -a
# -q参数，只显示container id
docker ps -q
# 查看容器详细信息
docker inspect [NAME]

  6. rename

  7. stats

  8. top

  9. wait

  10. cp：拷贝文件

    #从容器外到容器内
#docker cp '容器外路径' '容器id:容器内路径'
docker cp /root/test.txt MyUbuntu:/tmp

  11. diff

  12. update

# 容器资源管理

  1. volume
  2. network

# 系统日志

  1. events
  2. history
  3. logs
